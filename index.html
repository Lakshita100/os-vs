program1.c Multi-Process Application using fork() 
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main()
{
    pid_t pid;
    pid = fork();

    if (pid < 0)
    {
        printf("Process creation failed\n");
    }
    else if (pid == 0)
    {
        printf("Child Process\n");
        printf("Child PID: %d\n", getpid());
    }
    else
    {
        wait(NULL);
        printf("Parent Process\n");
        printf("Parent PID: %d\n", getpid());
    }
    return 0;
}
program2.c Multi-Threaded Application using POSIX Threads 
#include <stdio.h>
#include <pthread.h>
#include <stdint.h> // Required for intptr_t

void* thread_function(void* arg)
{
    intptr_t id = (intptr_t)arg;
    printf("Thread %ld is running\n", (long)id);
    return NULL;
}

int main()
{
    pthread_t threads[3];
    intptr_t i; // Match the type used in the thread function
    for (i = 0; i < 3; i++)
    {
        pthread_create(&threads[i], NULL, thread_function, (void*)i);
    }
    for (i = 0; i < 3; i++)
    {
        pthread_join(threads[i], NULL);
    }
    return 0;
}
program3.c Dining Philosophers Problem 
#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>
#include <stdint.h>

sem_t room;
sem_t chopstick[5];

void eat(int phil) {
    printf("Philosopher %d is eating\n", phil);
}

void* philosopher(void* num) {
    int phil = (int)(intptr_t)num;

    sem_wait(&room);
    printf("Philosopher %d has entered the room\n", phil);

    sem_wait(&chopstick[phil]);
    sem_wait(&chopstick[(phil + 1) % 5]);

    eat(phil);
    sleep(2);

    printf("Philosopher %d has finished eating\n", phil);

    sem_post(&chopstick[(phil + 1) % 5]);
    sem_post(&chopstick[phil]);
    sem_post(&room);

    return NULL;
}

int main() {
    int i;
    pthread_t tid[5];

    sem_init(&room, 0, 4);

    for (i = 0; i < 5; i++) {
        sem_init(&chopstick[i], 0, 1);
    }

    for (i = 0; i < 5; i++) {
        pthread_create(&tid[i], NULL, philosopher, (void*)(intptr_t)i);
    }

    for (i = 0; i < 5; i++) {
        pthread_join(tid[i], NULL);
    }

    return 0;
}
program4.c Banker's Algorithm 
#include<stdio.h>

int main()
{
  int Max[10][10], need[10][10], alloc[10][10], avail[10], completed[10], safeSequence[10];
  int p, r, i, j, process, count;
  count = 0;

  printf("Enter the Number of Process: ");
  scanf("%d", &p);

  for(i = 0; i < p; i++)
  {
    completed[i] = 0;
  }

  printf("\nEnter the no of resources: ");
  scanf("%d", &r);

  printf("\nEnter the Max Matrix for each process : ");
  for (i = 0; i < p; i++)
  {
    printf("\nFor process %d :", i + 1);
    for(j = 0; j < r; j++)
      scanf("%d", &Max[i][j]);
  }

  printf("\nEnter the allocation for Each Process: ");
  for(i = 0; i < p; i++)
  {
    printf("\nFor process %d :", i + 1);
    for(j = 0; j < r; j++)
      scanf("%d", &alloc[i][j]);
  }

  printf("\nEnter the Available Resources: ");
  for (i = 0; i < r; i++)
    scanf("%d", &avail[i]);

  for (i = 0; i < p; i++)
    for(j = 0; j < r; j++)
      need[i][j] = Max[i][j] - alloc[i][j];

  do
  {
    printf("\nMax matrix:\tAllocation matrix:\n");
    for(i = 0; i < p; i++)
    {
      for(j = 0; j < r; j++)
        printf("%d ", Max[i][j]);
      printf("\t");
      for(j = 0; j < r; j++)
        printf("%d ", alloc[i][j]);
      printf("\n");
    }

    process = -1;
    for(i = 0; i < p; i++)
    {
      if(completed[i] == 0)
      {
        process = i;
        for(j = 0; j < r; j++)
        {
          if(avail[j] < need[i][j])
          {
            process = -1;
            break;
          }
        }
      }

      if(process != -1)
      {
        printf("\nProcess %d runs to completion !", process + 1);
        safeSequence[count] = process + 1;
        count++;
        for(j = 0; j < r; j++)
        {
          avail[j] += alloc[process][j];
          alloc[process][j] = 0;
          Max[process][j] = 0;
          completed[process] = -1;
        }
        break;
      }
    }
  } while(count != p && process != -1);

  if(count == p)
  {
    printf("\nThe system is in a safe state!!\n");
    printf("Safe Sequence: < ");
    for(i = 0; i < p; i++)
      printf("%d ", safeSequence[i]);
    printf(">\n");
  }
  else
  {
    printf("\nThe system is in an unsafe state!!");
  }
  return 0;
}
program5.c FIFO Page Replacement 
#include <stdio.h>

int main() {
    int frames, pages, page_faults = 0;
    printf("Enter number of frames: ");
    scanf("%d", &frames);
    printf("Enter number of pages: ");
    scanf("%d", &pages);

    int page_string[pages], queue[frames], front = 0, count = 0;

    printf("Enter the reference string: ");
    for (int i = 0; i < pages; i++)
        scanf("%d", &page_string[i]);

    for (int i = 0; i < pages; i++) {
        int found = 0;
        for (int j = 0; j < count; j++) {
            if (queue[j] == page_string[i]) {
                found = 1;
                break;
            }
        }

        if (!found) {
            if (count < frames)
                queue[count++] = page_string[i];
            else {
                queue[front] = page_string[i];
                front = (front + 1) % frames;
            }
            page_faults++;
        }

        printf("\nFrames: ");
        for (int k = 0; k < count; k++)
            printf("%d ", queue[k]);
    }

    printf("\nTotal Page Faults: %d\n", page_faults);
    return 0;
}
program6.c LRU Page Replacement 
#include <stdio.h>

int findLRU(int time[], int n) {
    int min = time[0], pos = 0;
    for (int i = 1; i < n; i++) {
        if (time[i] < min) {
            min = time[i];
            pos = i;
        }
    }
    return pos;
}

int main() {
    int frames, pages, page_faults = 0;

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    printf("Enter number of pages: ");
    scanf("%d", &pages);

    int page_string[pages], queue[frames], time[frames], counter = 0;

    printf("Enter the reference string: ");
    for (int i = 0; i < pages; i++)
        scanf("%d", &page_string[i]);

    for (int i = 0; i < frames; i++) {
        queue[i] = -1;
        time[i] = -1;
    }

    for (int i = 0; i < pages; i++) {
        int found = 0;

        for (int j = 0; j < frames; j++) {
            if (queue[j] == page_string[i]) {
                found = 1;
                time[j] = counter++;
                break;
            }
        }

        if (!found) {
            int pos = -1;

            for (int j = 0; j < frames; j++) {
                if (queue[j] == -1) {
                    pos = j;
                    break;
                }
            }

            if (pos == -1)
                pos = findLRU(time, frames);

            queue[pos] = page_string[i];
            time[pos] = counter++;
            page_faults++;
        }

        printf("\nFrames: ");
        for (int k = 0; k < frames; k++)
            printf("%d ", queue[k]);
    }

    printf("\nTotal Page Faults: %d\n", page_faults);
    return 0;
}
program7.c First Fit Memory Allocation 
#include <stdio.h>

int main() {
    int nB, nP;

    printf("Enter number of memory blocks: ");
    scanf("%d", &nB);
    int block[nB];
    printf("Enter sizes of memory blocks:\n");
    for (int i = 0; i < nB; i++) scanf("%d", &block[i]);

    printf("Enter number of processes: ");
    scanf("%d", &nP);
    int proc[nP];
    printf("Enter sizes of processes:\n");
    for (int i = 0; i < nP; i++) scanf("%d", &proc[i]);

    int alloc[nP], frag[nP];
    for (int i = 0; i < nP; i++) { alloc[i] = -1; frag[i] = 0; }

    // FIRST FIT
    for (int i = 0; i < nP; i++) {
        for (int j = 0; j < nB; j++) {
            if (block[j] >= proc[i]) {
                alloc[i] = j;
                frag[i] = block[j] - proc[i];
                block[j] -= proc[i];
                break;
            }
        }
    }

    printf("\n----- FIRST FIT ALLOCATION -----\n");
    printf("Process\tSize\tBlock\tFragment\n");
    for (int i = 0; i < nP; i++) {
        printf("P%d\t%d\t", i + 1, proc[i]);
        if (alloc[i] == -1) printf("NA\t-\n");
        else printf("%d\t%d\n", alloc[i] + 1, frag[i]);
    }

    return 0;
}

program10.c FCFS Disk Scheduling 
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, i;
    int request[50];
    int head;
    int total_seek = 0;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter the request sequence:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &request[i]);
    }

    for(i = 0; i < n; i++) {
        total_seek += abs(head - request[i]);
        head = request[i];
    }

    printf("Total Seek Time (FCFS) = %d\n", total_seek);

    return 0;
}
program11.c SSTF Disk Scheduling 
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

int main() {
    int n, i, count = 0;
    int req[50], visited[50] = {0};
    int head, total = 0;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter request sequence:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &req[i]);
    }

    while (count < n) {
        int min = INT_MAX, index = -1;
        for (i = 0; i < n; i++) {
            if (visited[i] == 0) {
                int dist = abs(head - req[i]);
                if (dist < min) {
                    min = dist;
                    index = i;
                }
            }
        }

        if (index == -1) {
            break;
        }

        total += min;
        head = req[index];
        visited[index] = 1;
        count++;
    }

    printf("Total Seek Time (SSTF) = %d\n", total);
    return 0;
}
